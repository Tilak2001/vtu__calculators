using System;

class Program
{
    static void Main(string[] args)
    {
        // Initial money in rupees
        double money = 50;

        // Cost of one chocolate
        double chocolateCost = 2;

        // Cashback for one chocolate wrapper
        double cashbackPerWrapper = 0.5;

        // Total chocolates bought
        int totalChocolates = 0;

        // Step 1: Buy as many chocolates as possible with initial money
        int chocolatesBought = (int)(money / chocolateCost);
        totalChocolates += chocolatesBought;

        // Step 2: Use the wrappers for cashback and repeat the process
        int wrappers = chocolatesBought;

        while (wrappers > 0)
        {
            double cashback = wrappers * cashbackPerWrapper;

            money = (money % chocolateCost) + cashback; 

            // Buy more chocolates with the new total money
            int newChocolates = (int)(money / chocolateCost);
            totalChocolates += newChocolates;

            // Update the remaining money after buying new chocolates
            money = money % chocolateCost;

            // Update wrappers count
            wrappers = newChocolates;
        }

        // Output the total number of chocolates you can buy
        Console.WriteLine("Total chocolates you can buy: " + totalChocolates);
    }
}
